def scatter_1k(data1, data2): # function to show scatterplot
    yes_1ka = yes_1k[[data1, 'Player']]
    not_1ka = not_1k[[data1, 'Player']]
    plt.figure(figsize = (5,5))
    plt.scatter(not_1ka[data1], not_1ka['Player'], color= 'red' )
    plt.scatter(yes_1ka[data1], yes_1ka['Player'], color= 'blue' )
    plt.title(f'Blue Dot Represents Player Future 1k Season Prediction\n Rookie Year {data2}')
    plt.xlabel(data2)
    plt.ylabel("Player Name")
    plt.show()
